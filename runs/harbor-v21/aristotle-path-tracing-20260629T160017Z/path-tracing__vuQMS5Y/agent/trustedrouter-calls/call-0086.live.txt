    next(f); wh=next(f).split(); w_ref,h_ref=int(wh[0]),int(wh[1]); next(f)
    ref = list(map(int, f.read().split()))

with open('/app/reconstructed.ppm') as f:
    next(f); wh=next(f).split(); w_rec,h_rec=int(wh[0]),int(wh[1]); next(f)
    rec = list(map(int, f.read().split()))

w, h = w_ref, h_ref

# Compute differences separately for sky and floor
sky_diff_sum = 0
sky_count = 0
floor_diff_sum = 0
floor_count = 0
sphere_diff_sum = 0
sphere_count = 0

for y in range(h):
    for x in range(w):
        i = (y*w + x)*3
        rr, gg, bb = ref[i], ref[i+1], ref[i+2]
        # Check if reference pixel is sky (B=255, R>100)
        if bb == 255 and rr > 100 and gg > 100:
            d = (rr-rec[i])**2 + (gg-rec[i+1])**2 + (bb-rec[i+2])**2
            sky_diff_sum += d
            sky_count += 1
        # Check if it's floor (R==G==B and in [67,152])
        elif rr == gg == bb and (rr == 67 or rr == 152):
            d = (rr-rec[i])**2 + (gg-rec[i+1])**2 + (bb-rec[i+2])**2
            floor_diff_sum += d
            floor_count += 1
        # Otherwise it's sphere/shadow
        else:
            d = (rr-rec[i])**2 + (gg-rec[i+1])**2 + (bb-rec[i+2])**2
            sphere_diff_sum += d
            sphere_count += 1

total = sky_count + floor_count + sphere_count
print(f"Sky: {100*sky_count/total:.1f}% pixels, diff={sky_diff_sum:.0f}, MSE={sky_diff_sum/(sky_count*3):.1f}")
print(f"Floor: {100*floor_count/total:.1f}% pixels, diff={floor_diff_sum:.0f}, MSE={floor_diff_sum/(floor_count*3):.1f}")
print(f"Sphere: {100*sphere_count/total:.1f}% pixels, diff={sphere_diff_sum:.0f}, MSE={sphere_diff_sum/(sphere_count*3):.1f}")
print(f"Total: {sky_diff_sum+floor_diff_sum+sphere_diff_sum:.0f}")
