print(f"oc = ({oc_x:.6f}, {oc_y:.6f}, {oc_z:.6f})")

# Solve for intersection
a = dx*dx + dy*dy + dz*dz  # should be 1
b = 2 * (oc_x*dx + oc_y*dy + oc_z*dz)
c = oc_x*oc_x + oc_y*oc_y + oc_z*oc_z - R*R

disc = b*b - 4*a*c
print(f"disc = {disc:.10f} (should be ~0 for tangent)")

if disc >= 0:
    t = (-b - math.sqrt(disc)) / (2*a)
    print(f"t = {t:.6f}")
    pt = (cam[0] + t*dx, cam[1] + t*dy, cam[2] + t*dz)
    print(f"intersection point: ({pt[0]:.6f}, {pt[1]:.6f}, {pt[2]:.6f})")
    # Normal at intersection
    n_x = (pt[0] - Xs)
    n_y = (pt[1] - Ys)
    n_z = (pt[2] - Zs)
    n_len = math.sqrt(n_x*n_x + n_y*n_y + n_z*n_z)
    n_x /= n_len
    n_y /= n_len
    n_z /= n_len
    print(f"normal: ({n_x:.6f}, {n_y:.6f}, {n_z:.6f})")
    
    # Light direction
    L = (5 - pt[0], 5 - pt[1], 5 - pt[2])
    L_len = math.sqrt(L[0]*L[0] + L[1]*L[1] + L[2]*L[2])
    L = (L[0]/L_len, L[1]/L_len, L[2]/L_len)
    print(f"L: ({L[0]:.6f}, {L[1]:.6f}, {L[2]:.6f})")
    
    diff = n_x*L[0] + n_y*L[1] + n_z*L[2]
    print(f"diff = {diff:.6f}")
    
    # Specular
    V = (-dx, -dy, -dz)
    H = (L[0]+V[0], L[1]+V[1], L[2]+V[2])
    H_len = math.sqrt(H[0]*H[0] + H[1]*H[1] + H[2]*H[2])
    H = (H[0]/H_len, H[1]/H_len, H[2]/H_len)
    spec = max(0, n_x*H[0] + n_y*H[1] + n_z*H[2])
    spec = spec**20
    
    r = min(255, (1.0 * (0.2 + 0.8 * max(0, diff)) * 255 + spec * 200))
    g = min(255, (0.2 * (0.2 + 0.8 * max(0, diff)) * 255 + spec * 40))
    b = min(255, (0.2 * (0.2 + 0.8 * max(0, diff)) * 255 + spec * 40))
    print(f"Computed color: ({r:.0f}, {g:.0f}, {b:.0f})")
    print(f"Reference color: (148, 29, 29)")
